Securing User Data: Removing Passwords from User Objects

Never include passwords—hashed or otherwise—in responses or tokens. Strip them before signing JWTs and before sending user objects back to the client.

Define the Schema Method


// models/user.js
const mongoose = require('mongoose')
const bcrypt = require('bcrypt')
const { Schema } = mongoose

const userSchema = new Schema({
  username: { type: String, required: true, lowercase: true, unique: true },
  password: { type: String, required: true },
  memberSince: { type: Date, default: Date.now },
  isAdmin: { type: Boolean, default: false }
})

// Remove password before sending user outward
userSchema.methods.withoutPassword = function () {
  const user = this.toObject()
  delete user.password
  return user
}

// (from prior lesson) Compare login attempt
userSchema.methods.checkPassword = async function (passwordAttempt) {
  return bcrypt.compare(passwordAttempt, this.password)
}

module.exports = mongoose.model('User', userSchema)

Use in /login Route


// routes/authRouter.js (excerpt)
const express = require('express')
const jwt = require('jsonwebtoken')
const User = require('../models/user')
const authRouter = express.Router()

authRouter.post('/login', async (req, res, next) => {
  try {
    const { username, password } = req.body
    const user = await User.findOne({ username: username.toLowerCase() })
    if (!user) { res.status(403); return next(new Error('Incorrect Username or Password')) }

    const isMatch = await user.checkPassword(password)
    if (!isMatch) { res.status(403); return next(new Error('Incorrect Username or Password')) }

    const safeUser = user.withoutPassword()
    const token = jwt.sign(safeUser, process.env.SECRET)
    return res.status(200).send({ token, user: safeUser })
  } catch (err) {
    res.status(500)
    return next(err)
  }
})

Use in /signup Route


// routes/authRouter.js (excerpt)
authRouter.post('/signup', async (req, res, next) => {
  try {
    const exists = await User.findOne({ username: req.body.username.toLowerCase() })
    if (exists) { res.status(403); return next(new Error('Username is already taken')) }

    const newUser = new User(req.body)
    const saved = await newUser.save()
    const safeUser = saved.withoutPassword()
    const token = jwt.sign(safeUser, process.env.SECRET)
    return res.status(201).send({ token, user: safeUser })
  } catch (err) {
    res.status(500)
    return next(err)
  }
})

Understanding Mongoose Method for Removing Password Field

Define a reusable instance method to omit sensitive fields before returning documents to clients or signing tokens.

Method Definition


userSchema.methods.withoutPassword = function () {
  const user = this.toObject()
  delete user.password
  return user
}

Why it Matters

  • Prevents Exposure: Keeps even hashed passwords out of responses/tokens.
  • Reduces Attack Surface: Transmit only what the client needs.
  • Best Practice: Centralize sanitization logic on the model.
