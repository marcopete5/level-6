Verifying Passwords with bcrypt in Token-Based Authentication

When a user logs in, compare the plaintext password attempt to the stored hashed password using bcrypt.compare. Expose this as an instance method on your User schema.

Schema Method (async/await)


// userSchema.js
const mongoose = require('mongoose')
const bcrypt = require('bcrypt')
const { Schema } = mongoose

const userSchema = new Schema({
  username: { type: String, required: true, lowercase: true, unique: true },
  password: { type: String, required: true },
  memberSince: { type: Date, default: Date.now },
  isAdmin: { type: Boolean, default: false }
})

// Compare a login attempt with the stored hash
userSchema.methods.checkPassword = async function (passwordAttempt) {
  try {
    return bcrypt.compare(passwordAttempt, this.password)
  } catch (error) {
    throw error
  }
}

module.exports = mongoose.model('User', userSchema)

Using the Method in a Login Route


// authRouter.js (excerpt)
const express = require('express')
const jwt = require('jsonwebtoken')
const User = require('../models/user.js')
const authRouter = express.Router()

authRouter.post('/login', async (req, res, next) => {
  try {
    const { username, password } = req.body
    const user = await User.findOne({ username: username.toLowerCase() })
    if (!user) {
      res.status(403)
      return next(new Error('Incorrect Username or Password'))
    }

    const isMatch = await user.checkPassword(password)
    if (!isMatch) {
      res.status(403)
      return next(new Error('Incorrect Username or Password'))
    }

    // Optionally strip password before sending user back
    const userObj = user.toObject()
    delete userObj.password

    const token = jwt.sign(userObj, process.env.SECRET)
    return res.status(200).send({ token, user: userObj })
  } catch (err) {
    res.status(500)
    return next(err)
  }
})

module.exports = authRouter

Schema Methods

Add reusable, model-specific behavior directly on documents. In this unit we use two: checkPassword (password verification) and withoutPassword (omit password before sending to clients).

Callback-style Variant (equivalent)


// Callback version
userSchema.methods.checkPassword = function (passwordAttempt, callback) {
  bcrypt.compare(passwordAttempt, this.password, (err, isMatch) => {
    if (err) return callback(err)
    return callback(null, isMatch)
  })
}
