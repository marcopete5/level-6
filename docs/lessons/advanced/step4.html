Handling Login/Signup Errors

Lesson

Learning Objectives

  • Show login and signup errors on the form to inform users.
  • Conditionally render either the login or signup form.

Lesson

Open in Browser

Why Display Errors (and why keep them vague)

  • UX: Immediate feedback helps users correct input and build trust.
  • Security: Vague messages prevent username/password enumeration and reduce brute-force hints.
  • Support: Clear, actionable guidance reduces support load and drop-off.

Implementation Examples


// In your UserContext.jsx
function handleAuthErr(errMsg) {
  setUserState(prev => ({ ...prev, errMsg }))
}

function resetAuthErr() {
  setUserState(prev => ({ ...prev, errMsg: "" }))
}


// In your AuthForm.jsx
export default function AuthForm({ submit, btnText, errMsg }) {
  const [inputs, setInputs] = React.useState({ username: "", password: "" })

  function handleChange(e) {
    const { name, value } = e.target
    setInputs(prev => ({ ...prev, [name]: value }))
  }

  function handleSubmit(e) {
    e.preventDefault()
    submit(inputs)
  }

  return (
    <form onSubmit={handleSubmit} className="card">
      <label>Username</label>
      <input name="username" value={inputs.username} onChange={handleChange} />

      <label>Password</label>
      <input type="password" name="password" value={inputs.password} onChange={handleChange} />

      {errMsg && <p style={{ color: "red" }} role="alert">{errMsg}</p>}
      <button className="btn-primary">{btnText}</button>
    </form>
  )
}


// In your Auth.jsx (container toggling forms)
import { useUser } from "../context/UserContext"
import AuthForm from "./AuthForm"

export default function Auth() {
  const { signup, login, userState: { errMsg }, resetAuthErr } = useUser()
  const [isMember, setIsMember] = React.useState(true)

  function toggleForm() {
    setIsMember(prev => !prev)
    resetAuthErr() // clear stale errors when switching
  }

  return (
    <div className="grid two-col gap-xl">
      {isMember ? (
        <AuthForm submit={login} btnText="Log In" errMsg={errMsg} />
      ) : (
        <AuthForm submit={signup} btnText="Sign Up" errMsg={errMsg} />
      )}

      <button className="btn-link" onClick={toggleForm}>
        {isMember ? "Need an account? Sign up" : "Already have an account? Log in"}
      </button>
    </div>
  )
}


// On the server (auth routes), prefer vague messages
if (!user || !(await user.checkPassword(req.body.password))) {
  res.status(403)
  return next(new Error("Incorrect Username or Password"))
}
